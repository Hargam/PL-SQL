-- parametrer le systeme pour nouveau utilisateur
alter session set "_ORACLE_SCRIPT"=true;

-- script pour nouveau utilisateur
-- USER SQL
CREATE USER "HR" IDENTIFIED BY "hr"
DEFAULT TABLESPACE "SYSAUX"
TEMPORARY TABLESPACE "TEMP";

-- QUOTAS
ALTER USER "HR" QUOTA UNLIMITED ON "SYSAUX";

-- ROLES
GRANT "RESOURCE" TO "HR" ;
ALTER USER "HR" DEFAULT ROLE "RESOURCE";

-- SYSTEM PRIVILEGES
GRANT CREATE TRIGGER TO "HR" ;
GRANT CREATE SESSION TO "HR" ;
GRANT CREATE TABLE TO "HR" ;
GRANT CREATE TABLESPACE TO "HR" ;
GRANT CREATE SEQUENCE TO "HR" ;
GRANT CREATE USER TO "HR" ;
GRANT UNLIMITED TABLESPACE TO "HR" ;
GRANT CREATE PROCEDURE TO "HR" ;
